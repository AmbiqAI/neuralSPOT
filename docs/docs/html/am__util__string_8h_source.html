57// Character attributes lookup table defines.
58//
59//*****************************************************************************
60#define AM_CATTR_NONE 0x00
61#define AM_CATTR_ALPHA 0x01
62#define AM_CATTR_LOWER 0x02
63#define AM_CATTR_UPPER 0x04
64#define AM_CATTR_DIGIT 0x08
65#define AM_CATTR_XDIGIT 0x10
66#define AM_CATTR_WHSPACE 0x20
67#define AM_CATTR_FILENM83 0x80
68
69//
70// Set MINIMIZE_CATTR_TABLE to 1 to configure for minimal CATTR table size,
71// (256 instead of 512 bytes) but at a cost of slightly larger code size.
72// However, setting this option also provides an additional level of checking
73// of the argument; if the argument is not a uint8_t, the functions are
74// guaranteed to return 0.
75//
76#define MINIMIZE_CATTR_TABLE 0

109//
110// By default all of the "is" functions are implemented as macros. To implement
111// as functions rather than macros, use a global compiler command line (-D)
112// option to define AM_UTIL_STRING_CTYPE_DISABLE_MACROS.
113//
114#ifdef AM_UTIL_STRING_CTYPE_DISABLE_MACROS
115extern int am_util_string_isalnum(int c);

118extern int am_util_string_islower(int c);
119extern int am_util_string_isspace(int c);
120extern int am_util_string_isupper(int c);
121extern int am_util_string_isxdigit(int c);
122extern int am_util_string_tolower(int c);

124
125// Non-standard "is" Functions
126extern int am_util_string_isfilenm83(int c);
127#else
128#if MINIMIZE_CATTR_TABLE
129#define am_util_string_isalnum(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & (AM_CATTR_ALPHA | AM_CATTR_DIGIT)) ? 1 : 0)
130#define am_util_string_isalpha(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & AM_CATTR_ALPHA) ? 1 : 0)
131#define am_util_string_isdigit(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & AM_CATTR_DIGIT) ? 1 : 0)
132#define am_util_string_islower(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & AM_CATTR_LOWER) ? 1 : 0)
133#define am_util_string_isspace(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & AM_CATTR_WHSPACE) ? 1 : 0)
134#define am_util_string_isupper(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & AM_CATTR_UPPER) ? 1 : 0)
135#define am_util_string_isxdigit(c) ((c & 0xffffff80) ? 0 : (am_cattr[c] & AM_CATTR_XDIGIT) ? 1 : 0)
136#define am_util_string_tolower(c) ((am_cattr[c & 0x7f] & AM_CATTR_UPPER) ? c | 0x20 : c)
137#define am_util_string_toupper(c) ((am_cattr[c & 0x7f] & AM_CATTR_LOWER) ? c & ~0x20 : c)
138#else
139#define am_util_string_isalnum(c) (am_cattr[c & 0xff] & (AM_CATTR_ALPHA | AM_CATTR_DIGIT))
140#define am_util_string_isalpha(c) (am_cattr[c & 0xff] & AM_CATTR_ALPHA)
141#define am_util_string_isdigit(c) (am_cattr[c & 0xff] & AM_CATTR_DIGIT)
142#define am_util_string_islower(c) (am_cattr[c & 0xff] & AM_CATTR_LOWER)
143#define am_util_string_isspace(c) (am_cattr[c & 0xff] & AM_CATTR_WHSPACE)
144#define am_util_string_isupper(c) (am_cattr[c & 0xff] & AM_CATTR_UPPER)
145#define am_util_string_isxdigit(c) (am_cattr[c & 0xff] & AM_CATTR_XDIGIT)
146#define am_util_string_tolower(c) ((am_cattr[c & 0xff] & AM_CATTR_UPPER) ? c | 0x20 : c)
147#define am_util_string_toupper(c) ((am_cattr[c & 0xff] & AM_CATTR_LOWER) ? c & ~0x20 : c)
148#endif // MINIMIZE_CATTR_TABLE
