83// Returns true if the queue is empty.

85#define am_hal_queue_empty(psQueue) \
86 ((psQueue)->ui32Length == 0)

88//
89// Returns true if the queue is full.
90//
91#define am_hal_queue_full(psQueue) \
92 ((psQueue)->ui32Length == (psQueue)->ui32Capacity)
93
94//
95// Returns the amount of space left in the queue (in bytes).
96//
97#define am_hal_queue_space_left(psQueue) \
98 ((psQueue)->ui32Capacity - (psQueue)->ui32Length)
99
100//
101// Returns the number of configured items that will fit in the queue.
102//
103#define am_hal_queue_slots_left(psQueue) \
104 (((psQueue)->ui32Capacity - (psQueue)->ui32Length) \
105 / (psQueue)->ui32ItemSize)
106
107//
108// Returns the amount of data in the queue (in bytes).
109//
110#define am_hal_queue_data_left(psQueue) \
111 ((psQueue)->ui32Length)
112
113//
114// Returns the number of configured items left in the queue.
115//
116#define am_hal_queue_items_left(psQueue) \
117 ((psQueue)->ui32Length / (psQueue)->ui32ItemSize)
118
119//
120// Can be used as a pointer to the next item to be read from the queue.
121//
122#define am_hal_queue_peek(psQueue) \
123 ((void *) &((psQueue)->pui8Data[(psQueue)->ui32ReadIndex]))
124
125//
126// Can be used as a pointer to the next available slot in the queue memory.
127//
128#define am_hal_queue_next_slot(psQueue) \
129 ((void *) &((psQueue)->pui8Data[(psQueue)->ui32WriteIndex]))
