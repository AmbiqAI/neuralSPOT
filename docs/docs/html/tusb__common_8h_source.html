99TU_ATTR_ALWAYS_INLINE static inline uint32_t tu_u32(uint8_t b3, uint8_t b2, uint8_t b1, uint8_t b0)
100{
101 return ( ((uint32_t) b3) << 24) | ( ((uint32_t) b2) << 16) | ( ((uint32_t) b1) << 8) | b0;
102}
103
104TU_ATTR_ALWAYS_INLINE static inline uint16_t tu_u16(uint8_t high, uint8_t low)
105{
106 return (uint16_t) ((((uint16_t) high) << 8) | low);
107}

192#elif TUP_MCU_STRICT_ALIGN
193
194// MCU such as LPC_IP3511 Highspeed cannot access unaligned memory on USB_RAM although it is ARM M4.
195// We have to manually pick up bytes since tu_unaligned_uint32_t will still generate unaligned code
196// NOTE: volatile cast to memory to prevent compiler to optimize and generate unaligned code
197// TODO Big Endian may need minor changes
198TU_ATTR_ALWAYS_INLINE static inline uint32_t tu_unaligned_read32(const void* mem)
199{
200 volatile uint8_t const* buf8 = (uint8_t const*) mem;
201 return tu_u32(buf8[3], buf8[2], buf8[1], buf8[0]);
202}
203
204TU_ATTR_ALWAYS_INLINE static inline void tu_unaligned_write32(void* mem, uint32_t value)
205{
206 volatile uint8_t* buf8 = (uint8_t*) mem;
207 buf8[0] = tu_u32_byte0(value);
208 buf8[1] = tu_u32_byte1(value);
209 buf8[2] = tu_u32_byte2(value);
210 buf8[3] = tu_u32_byte3(value);
211}
212
213TU_ATTR_ALWAYS_INLINE static inline uint16_t tu_unaligned_read16(const void* mem)
214{
215 volatile uint8_t const* buf8 = (uint8_t const*) mem;
216 return tu_u16(buf8[1], buf8[0]);
217}
218
219TU_ATTR_ALWAYS_INLINE static inline void tu_unaligned_write16(void* mem, uint16_t value)
220{
221 volatile uint8_t* buf8 = (uint8_t*) mem;
222 buf8[0] = tu_u16_low(value);
223 buf8[1] = tu_u16_high(value);
224}

358typedef struct
359{
360 uint32_t key;
361 const char* data;
362} tu_lookup_entry_t;
363
364typedef struct
365{
366 uint16_t count;
367 tu_lookup_entry_t const* items;
368} tu_lookup_table_t;
369
370static inline const char* tu_lookup_find(tu_lookup_table_t const* p_table, uint32_t key)
371{
372 static char not_found[11];
373
374 for(uint16_t i=0; i<p_table->count; i++)
375 {
376 if (p_table->items[i].key == key) return p_table->items[i].data;
377 }
378
379 // not found return the key value in hex
380 sprintf(not_found, "0x%08lX", (unsigned long) key);
381
382 return not_found;
383}
