499// Get direction from Endpoint address
500static inline tusb_dir_t tu_edpt_dir(uint8_t addr)
501{
502 return (addr & TUSB_DIR_IN_MASK) ? TUSB_DIR_IN : TUSB_DIR_OUT;
503}
504
505// Get Endpoint number from address
506static inline uint8_t tu_edpt_number(uint8_t addr)
507{
508 return (uint8_t)(addr & (~TUSB_DIR_IN_MASK));
509}
510
511static inline uint8_t tu_edpt_addr(uint8_t num, uint8_t dir)
512{
513 return (uint8_t)(num | (dir ? TUSB_DIR_IN_MASK : 0));
514}
515
516static inline uint16_t tu_edpt_packet_size(tusb_desc_endpoint_t const* desc_ep)
517{
518 return tu_le16toh(desc_ep->wMaxPacketSize) & TU_GENMASK(10, 0);
519}
520
521//--------------------------------------------------------------------+
522// Descriptor helper
523//--------------------------------------------------------------------+
524static inline uint8_t const * tu_desc_next(void const* desc)
525{
526 uint8_t const* desc8 = (uint8_t const*) desc;
527 return desc8 + desc8[DESC_OFFSET_LEN];
528}
529
530static inline uint8_t tu_desc_type(void const* desc)
531{
532 return ((uint8_t const*) desc)[DESC_OFFSET_TYPE];
533}
534
535static inline uint8_t tu_desc_len(void const* desc)
536{
537 return ((uint8_t const*) desc)[DESC_OFFSET_LEN];
538}
