39class MicroInterpreter {
40 public:
41 // The lifetime of the model, op resolver, tensor arena, error reporter,
42 // resource variables, and profiler must be at least as long as that of the
43 // interpreter object, since the interpreter may need to access them at any
44 // time. This means that you should usually create them with the same scope as
45 // each other, for example having them all allocated on the stack as local
46 // variables through a top-level function. The interpreter doesn't do any
47 // deallocation of any of the pointed-to objects, ownership remains with the
48 // caller.
49 MicroInterpreter(const Model* model, const MicroOpResolver& op_resolver,
50 uint8_t* tensor_arena, size_t tensor_arena_size,
51 ErrorReporter* error_reporter,
52 MicroResourceVariables* resource_variables = nullptr,
53 MicroProfiler* profiler = nullptr);
54
55 // Create an interpreter instance using an existing MicroAllocator instance.
56 // This constructor should be used when creating an allocator that needs to
57 // have allocation handled in more than one interpreter or for recording
58 // allocations inside the interpreter. The lifetime of the allocator must be
59 // as long as that of the interpreter object.
60 MicroInterpreter(const Model* model, const MicroOpResolver& op_resolver,
61 MicroAllocator* allocator, ErrorReporter* error_reporter,
62 MicroResourceVariables* resource_variables = nullptr,
63 MicroProfiler* profiler = nullptr);
64
65 ~MicroInterpreter();
66
67 // Runs through the model and allocates all necessary input, output and
68 // intermediate tensors.
69 TfLiteStatus AllocateTensors();
70
71 // In order to support partial graph runs for strided models, this can return
72 // values other than kTfLiteOk and kTfLiteError.
73 // TODO(b/149795762): Add this to the TfLiteStatus enum.
74 TfLiteStatus Invoke();
75
76 // This is the recommended API for an application to pass an external payload
77 // pointer as an external context to kernels. The life time of the payload
78 // pointer should be at least as long as this interpreter. TFLM supports only
79 // one external context.
80 TfLiteStatus SetMicroExternalContext(void* external_context_payload);
81
82 // This function is used by the TfLiteContext::GetExternalContext() to get the
83 // external context.
84 void* GetMicroExternalContext();
85
86 TfLiteTensor* input(size_t index);
87 size_t inputs_size() const {
88 return model_->subgraphs()->Get(0)->inputs()->size();
89 }
90 const flatbuffers::Vector<int32_t>& inputs() const {
91 return *model_->subgraphs()->Get(0)->inputs();
92 }
93 TfLiteTensor* input_tensor(size_t index) { return input(index); }
94 template <class T>
95 T* typed_input_tensor(int tensor_index) {
96 if (TfLiteTensor* tensor_ptr = input_tensor(tensor_index)) {
97 if (tensor_ptr->type == typeToTfLiteType<T>()) {
98 return GetTensorData<T>(tensor_ptr);
99 }
100 }
101 return nullptr;
102 }
103
104 TfLiteTensor* output(size_t index);
105 size_t outputs_size() const {
106 return model_->subgraphs()->Get(0)->outputs()->size();
107 }
108 const flatbuffers::Vector<int32_t>& outputs() const {
109 return *model_->subgraphs()->Get(0)->outputs();
110 }
111 TfLiteTensor* output_tensor(size_t index) { return output(index); }
112 template <class T>
113 T* typed_output_tensor(int tensor_index) {
114 if (TfLiteTensor* tensor_ptr = output_tensor(tensor_index)) {
115 if (tensor_ptr->type == typeToTfLiteType<T>()) {
116 return GetTensorData<T>(tensor_ptr);
117 }
118 }
119 return nullptr;
120 }
121
122 // Reset all variable tensors to the default value.
123 TfLiteStatus ResetVariableTensors();
124
125 TfLiteStatus initialization_status() const { return initialization_status_; }
126
127 // Populates node and registration pointers representing the inference graph
128 // of the model from values inside the flatbuffer (loaded from the TfLiteModel
129 // instance). Persistent data (e.g. operator data) is allocated from the
130 // arena.
131 TfLiteStatus PrepareNodeAndRegistrationDataFromFlatbuffer();
132
133 // For debugging only.
134 // Returns the actual used arena in bytes. This method gives the optimal arena
135 // size. It's only available after `AllocateTensors` has been called.
136 // Note that normally `tensor_arena` requires 16 bytes alignment to fully
137 // utilize the space. If it's not the case, the optimial arena size would be
138 // arena_used_bytes() + 16.
139 size_t arena_used_bytes() const { return allocator_.used_bytes(); }
140
141 protected:
142 const MicroAllocator& allocator() const { return allocator_; }
143 const TfLiteContext& context() const { return context_; }
144
145 private:
146 // TODO(b/158263161): Consider switching to Create() function to enable better
147 // error reporting during initialization.
148 void Init(MicroProfiler* profiler);
149
150 // Gets the current subgraph index used from within context methods.
151 int get_subgraph_index() { return graph_.GetCurrentSubgraphIndex(); }
152
153 // Static functions that are bound to the TfLiteContext instance:
154 static void* AllocatePersistentBuffer(TfLiteContext* ctx, size_t bytes);
155 static TfLiteStatus RequestScratchBufferInArena(TfLiteContext* ctx,
156 size_t bytes,
157 int* buffer_idx);
158 static void* GetScratchBuffer(TfLiteContext* ctx, int buffer_idx);
159 static void ReportOpError(struct TfLiteContext* context, const char* format,
160 ...);
161 static TfLiteTensor* GetTensor(const struct TfLiteContext* context,
162 int tensor_idx);
163 static TfLiteEvalTensor* GetEvalTensor(const struct TfLiteContext* context,
164 int tensor_idx);
165 static TfLiteStatus GetGraph(struct TfLiteContext* context,
166 TfLiteIntArray** args);
167
168 // This callback is an implementation for TfLiteContext::GetExternalContext
169 // interface.
170 static TfLiteExternalContext* GetExternalContext(
171 TfLiteContext* context, TfLiteExternalContextType unused);
172
173 const Model* model_;
174 const MicroOpResolver& op_resolver_;
175 ErrorReporter* error_reporter_;
176 TfLiteContext context_ = {};
177 MicroAllocator& allocator_;
178 MicroGraph graph_;
179 bool tensors_allocated_;
180
181 TfLiteStatus initialization_status_;
182
183 ScratchBufferHandle* scratch_buffer_handles_ = nullptr;
184 void* external_context_payload_ = nullptr;
185
186 // TODO(b/162311891): Clean these pointers up when this class supports buffers
187 // from TfLiteEvalTensor.
188 TfLiteTensor** input_tensors_;
189 TfLiteTensor** output_tensors_;
190};
