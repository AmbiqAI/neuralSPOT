46// Holds placeholder information for a scratch buffer request from a kernel.
47// This struct is only used during the model prepare stage. Each request from a
48// kernel is stored in the head section. During the prepare stage, the head
49// section will at least hold kMaxScratchBuffersPerOp number of requests plus
50// any requests from previous kernel requests.
51//
52// When the memory plan is finalized, these structs are no longer used in favor
53// of a sequential, array of ScratchBufferHandle allocations in the tail
54// section. These allocations are indexed by the request API defined in the
55// TfLiteContext struct.
56typedef struct {
57 // Number of bytes required by the buffer. The actual allocated size might be
58 // greater than `bytes` due to buffer alignment.
59 size_t bytes;
60 // Node where the buffer is allocated for. This provides useful information to
61 // determine the lifetime of the buffer. In AllocationInfo, this buffer will
62 // have `before` = node_idx and `after` = node_idx.
63 int node_idx;
64} ScratchBufferRequest;

66} // namespace internal

88// Allocator responsible for allocating memory for all intermediate tensors
89// necessary to invoke a model.
90//
91// The lifetime of the model, tensor arena and error reporter must be at
92// least as long as that of the allocator object, since the allocator needs
93// them to be accessible during its entire lifetime.
94//
95// The MicroAllocator simply plans out additional allocations that are required
96// to standup a model for inference in TF Micro. This class currently relies on
97// an additional allocator - SimpleMemoryAllocator - for all allocations from an
98// arena. These allocations are divided into head (non-persistent) and tail
99// (persistent) regions:
100//
101// Memory layout to help understand how it works
102// This information could change in the future version.
103// ************** .memory_allocator->GetBuffer()
104// Tensors/Scratch buffers (head)
105// ************** .head_watermark
106// unused memory
107// ************** .memory_allocator->GetBuffer() + ->GetMaxBufferSize()
108// - ->GetDataSize()
109// persistent area (tail)
110// ************** .memory_allocator->GetBuffer() + ->GetMaxBufferSize()
111class MicroAllocator {
112 public:
113 // Creates a MicroAllocator instance from a given tensor arena. This arena
114 // will be managed by the created instance. The GreedyMemoryPlanner will
115 // by default be used and created on the arena.
116 // Note: Please use alignas(16) to make sure tensor_arena is 16
117 // bytes aligned, otherwise some head room will be wasted.
118 // TODO(b/157615197): Cleanup constructor + factory usage.
119 static MicroAllocator* Create(uint8_t* tensor_arena, size_t arena_size,
120 ErrorReporter* error_reporter);
121
122 // Creates a MicroAllocator instance from a given tensor arena and a given
123 // MemoryPlanner. This arena will be managed by the created instance. Note:
124 // Please use alignas(16) to make sure tensor_arena is 16 bytes
125 // aligned, otherwise some head room will be wasted.
126 static MicroAllocator* Create(uint8_t* tensor_arena, size_t arena_size,
127 MicroMemoryPlanner* memory_planner,
128 ErrorReporter* error_reporter);
129
130 // Creates a MicroAllocator instance using the provided SimpleMemoryAllocator
131 // instance and the MemoryPlanner. This allocator instance will use the
132 // SimpleMemoryAllocator instance to manage allocations internally.
133 static MicroAllocator* Create(SimpleMemoryAllocator* memory_allocator,
134 MicroMemoryPlanner* memory_planner,
135 ErrorReporter* error_reporter);
136
137 // Returns the fixed amount of memory overhead of MicroAllocator.
138 static size_t GetDefaultTailUsage(bool is_memory_planner_given);
139
140 // Allocates internal resources required for model inference for each subgraph
141 // from the arena.
142 //
143 // This method will run through the flatbuffer data supplied in the model to
144 // properly allocate tensor, node, and op registration data. This method is
145 // expected to be followed with a call to FinishModelAllocation() Returns a
146 // pointer to an array of SubgraphAllocations (also stored in the tail of the
147 // arena) where each index corresponds to a different subgraph in the model.
148 // Return value is nullptr if the allocations failed.
149 SubgraphAllocations* StartModelAllocation(const Model* model);
150
151 // Finish allocating internal resources required for model inference.
152 //
153 // -Plan the memory for activation tensors and scratch buffers.
154 // -Update eval tensors for each subgraph based on planned offsets.
155 // -Allocate scratch buffer handles array and update based on planned offsets.
156 //
157 // This method should be called after assigning model resources
158 // in StartModelAllocation(). The subgraph_allocations pointer should be the
159 // value passed into this class during StartModelAllocation(). Scratch buffer
160 // handles are stored in the out-param `scratch_buffer_handles` array which is
161 // allocated in this method. This value will be used in `GetScratchBuffer`
162 // call to retrieve scratch buffers.
163 TfLiteStatus FinishModelAllocation(
164 const Model* model, SubgraphAllocations* subgraph_allocations,
165 ScratchBufferHandle** scratch_buffer_handles);
166
167 // Allocates a TfLiteTensor struct and populates the returned value with
168 // properties from the model flatbuffer. This struct is allocated from
169 // persistent arena memory is only guaranteed for the lifetime of the
170 // application. The eval_tensors pointer should be the value passed into this
171 // class during StartModelAllocation() and contains the source-of-truth for
172 // buffers.
173 virtual TfLiteTensor* AllocatePersistentTfLiteTensor(
174 const Model* model, const SubgraphAllocations* subgraph_allocations,
175 int tensor_index, int subgraph_index);
176
177 // Allocates a TfLiteTensor struct and populates the returned value with
178 // properties from the model flatbuffer. This struct is allocated from
179 // temporary arena memory is only guaranteed until a call is made to
180 // ResetTempAllocations(). Subgraph_allocaitons contains the array of
181 // TfLiteEvalTensors. If the newly allocated temp at the specified subgraph
182 // and tensor index is already present int the TfLiteEvalTensor array, its
183 // data buffer will be re-used.
184 virtual TfLiteTensor* AllocateTempTfLiteTensor(
185 const Model* model, const SubgraphAllocations* subgraph_allocations,
186 int tensor_index, int subgraph_index);
187
188 // Resets all temporary allocations. This method should be called after a
189 // chain of temp allocations (e.g. chain of TfLiteTensor objects via
190 // AllocateTfLiteTensor()).
191 virtual void ResetTempAllocations();
192
193 // Allocates persistent buffer which has the same life time as the allocator.
194 // The memory is immediately available and is allocated from the tail of the
195 // arena.
196 virtual void* AllocatePersistentBuffer(size_t bytes);
197
198 // Register a scratch buffer of size `bytes` for Node with `node_id`.
199 // This method only requests a buffer with a given size to be used after a
200 // model has finished allocation via FinishModelAllocation(). All requested
201 // buffers will be accessible by the out-param in that method.
202 TfLiteStatus RequestScratchBufferInArena(size_t bytes, int subgraph_idx,
203 int* buffer_idx);
204
205 // Finish allocating a specific NodeAndRegistration prepare block (kernel
206 // entry for a model) with a given node ID. This call ensures that any scratch
207 // buffer requests and temporary allocations are handled and ready for the
208 // next node prepare block.
209 TfLiteStatus FinishPrepareNodeAllocations(int node_id);
210
211 // Returns the arena usage in bytes, only available after
212 // `FinishModelAllocation`. Otherwise, it will return 0.
213 size_t used_bytes() const;
214
215 BuiltinDataAllocator* GetBuiltinDataAllocator();
216
217 protected:
218 MicroAllocator(SimpleMemoryAllocator* memory_allocator,
219 MicroMemoryPlanner* memory_planner,
220 ErrorReporter* error_reporter);
221 virtual ~MicroAllocator();
222
223 // Allocates an array in the arena to hold pointers to the node and
224 // registration pointers required to represent the inference graph of the
225 // model.
226 virtual TfLiteStatus AllocateNodeAndRegistrations(
227 const Model* model, SubgraphAllocations* subgraph_allocations);
228
229 // Allocates the list of persistent TfLiteEvalTensors that are used for the
230 // "eval" phase of model inference. These structs will be the source of truth
231 // for all tensor buffers.
232 virtual TfLiteStatus AllocateTfLiteEvalTensors(
233 const Model* model, SubgraphAllocations* subgraph_allocations);
234 // Allocates persistent tensor buffers for variable tensors in the subgraph.
235 virtual TfLiteStatus AllocateVariables(const SubGraph* subgraph,
236 TfLiteEvalTensor* eval_tensors);
237
238 // Allocate and return a persistent TfLiteTensor.
239 // TODO(b/162311891): Drop this method when the interpreter has an API for
240 // accessing TfLiteEvalTensor structs.
241 virtual TfLiteTensor* AllocatePersistentTfLiteTensorInternal();
242
243 // Populates a TfLiteTensor struct with data from the model flatbuffer. Any
244 // quantization data is allocated from either the tail (persistent) or temp
245 // sections of the arena based on the allocation flag.
246 virtual TfLiteStatus PopulateTfLiteTensorFromFlatbuffer(const Model* model,
247 TfLiteTensor* tensor,
248 int tensor_index,
249 int subgraph_idx,
250 bool allocate_temp);
251
252 ErrorReporter* error_reporter() const;
253
254 private:
255 // Commits a memory plan for all non-persistent buffer allocations in the
256 // 'head' section of the memory arena. The eval_tensors pointer is the list of
257 // pre-allocated TfLiteEvalTensor structs that will point to the buffers that
258 // will be allocated into the head section in this function call. The
259 // scratch_buffer_handles pointer is the array of pre-allocated
260 // ScratchBufferHandle structs that will point to allocated buffers also in
261 // the head section.
262 virtual TfLiteStatus CommitStaticMemoryPlan(
263 const Model* model, TfLiteEvalTensor* eval_tensors,
264 ScratchBufferHandle* scratch_buffer_handles, int subgraph_idx);
265
266 // Allocates an array of ScratchBufferHandle structs in the tail section for a
267 // given number of handles.
268 virtual TfLiteStatus AllocateScratchBufferHandles(
269 ScratchBufferHandle** scratch_buffer_handles, size_t handle_count);
270
271 // Clears all internal scratch buffer request counts and resets the head to
272 // prepare for kernels to request scratch buffer data when a model is
273 // preparing.
274 TfLiteStatus InitScratchBufferData();
275
276 // Returns the pointer for the array of ScratchBufferRequest allocations in
277 // the head section.
278 internal::ScratchBufferRequest* GetScratchBufferRequests();
279
280 // A simple memory allocator that always allocate from the arena tail or head.
281 SimpleMemoryAllocator* memory_allocator_;
282
283 // Allocator used to allocate persistent builtin data.
284 BuiltinDataAllocator* builtin_data_allocator_;
285
286 // Activation buffer memory planner.
287 MicroMemoryPlanner* memory_planner_;
288
289 ErrorReporter* error_reporter_;
290 bool model_is_allocating_;
291
292 // Holds the number of ScratchBufferRequest instances stored in the head
293 // section when a model is allocating.
294 size_t scratch_buffer_request_count_ = 0;
295
296 // Holds the byte length of the memory plan with the largest head usage. Used
297 // to ensure that multi-tenant allocations can share the head for buffers.
298 size_t max_head_buffer_usage_ = 0;
299
300 TF_LITE_REMOVE_VIRTUAL_DELETE
301};
