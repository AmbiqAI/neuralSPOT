26class RuntimeShape {

30 RuntimeShape() : size_(0) {}

32 explicit RuntimeShape(int dimensions_count) : size_(dimensions_count) {}
33
34 RuntimeShape(int shape_size, int32_t value) : size_(shape_size) {
35 for (int i = 0; i < shape_size; ++i) {
36 SetDim(i, value);
37 }
38 }

40 RuntimeShape(int dimensions_count, const int32_t* dims_data)
41 : size_(dimensions_count) {
42 ReplaceWith(dimensions_count, dims_data);
43 }
44
45 bool operator==(const RuntimeShape& comp) const {
46 return this->size_ == comp.size_ &&
47 std::memcmp(DimsData(), comp.DimsData(), size_ * sizeof(int32_t)) ==
48 0;
49 }
50
51 ~RuntimeShape() {}
52
53 int32_t DimensionsCount() const { return size_; }
54 int32_t Dims(int i) const {
55 TFLITE_DCHECK_GE(i, 0);
56 TFLITE_DCHECK_LT(i, size_);
57 return dims_[i];
58 }
59 void SetDim(int i, int32_t val) {
60 TFLITE_DCHECK_GE(i, 0);
61 TFLITE_DCHECK_LT(i, size_);
62 dims_[i] = val;
63 }

65 static RuntimeShape ExtendedShape(int new_shape_size,
66 const RuntimeShape& shape) {
67 return RuntimeShape(new_shape_size, shape, 1);
68 }
69 int32_t* DimsData() { return dims_; }
70 const int32_t* DimsData() const { return dims_; }
71 const int32_t* DimsDataUpTo5D() const { return dims_; }
72
73 void ReplaceWith(int dimensions_count, const int32_t* dims_data) {
74 size_ = dimensions_count;
75 int32_t* dst_dims = DimsData();
76 std::memcpy(dst_dims, dims_data, dimensions_count * sizeof(int32_t));
77 }

79 // Returns the total count of elements, that is the size when flattened into a
80 // vector.
81 int FlatSize() const {
82 int buffer_size = 1;
83 const int* dims_data = reinterpret_cast<const int*>(DimsData());
84 for (int i = 0; i < size_; i++) {
85 buffer_size *= dims_data[i];
86 }
87 return buffer_size;
88 }
89
90 private:
91 // For use only by ExtendedShape(), written to guarantee (return-value) copy
92 // elision in C++17.
93 // This creates a shape padded to the desired size with the specified value.
94 RuntimeShape(int new_shape_size, const RuntimeShape& shape, int pad_value)
95 : size_(new_shape_size) {
96 // If the following check fails, it is likely because a 4D-only kernel is
97 // being used with an array of larger dimension count.
98 TFLITE_CHECK_GE(new_shape_size, shape.DimensionsCount());
99 const int size_increase = new_shape_size - shape.DimensionsCount();
100 for (int i = 0; i < size_increase; ++i) {
101 SetDim(i, pad_value);
102 }
103 std::memcpy(DimsData() + size_increase, shape.DimsData(),
104 sizeof(int32_t) * shape.DimensionsCount());
105 }
106
107 // A maximum of 4 dimensions are supported on TFLM.
108 static constexpr int kMaxSize = 5;
109 int32_t size_;
110 union {
111 int32_t dims_[kMaxSize];
112 };
113};

115// Since tensors with '0' in their shape are valid in TF, these offset functions
116// allow that as long as the corresponding index is also 0. It is upto the
117// calling ops to ensure that they perform verification checks on tensor shapes
118// if they don't support a particular behavior.
119
120inline int Offset(const RuntimeShape& shape, int i0, int i1, int i2, int i3) {
121 TFLITE_DCHECK_EQ(shape.DimensionsCount(), 4);
122 const int* dims_data = reinterpret_cast<const int*>(shape.DimsData());
123 TFLITE_DCHECK((dims_data[0] == 0 && i0 == 0) ||
124 (i0 >= 0 && i0 < dims_data[0]));
125 TFLITE_DCHECK((dims_data[1] == 0 && i1 == 0) ||
126 (i1 >= 0 && i1 < dims_data[1]));
127 TFLITE_DCHECK((dims_data[2] == 0 && i2 == 0) ||
128 (i2 >= 0 && i2 < dims_data[2]));
129 TFLITE_DCHECK((dims_data[3] == 0 && i3 == 0) ||
130 (i3 >= 0 && i3 < dims_data[3]));
131 return ((i0 * dims_data[1] + i1) * dims_data[2] + i2) * dims_data[3] + i3;
132}
133
134inline int Offset(const RuntimeShape& shape, int i0, int i1, int i2, int i3,
135 int i4) {
136 TFLITE_DCHECK_EQ(shape.DimensionsCount(), 5);
137 const int* dims_data = reinterpret_cast<const int*>(shape.DimsData());
138 TFLITE_DCHECK((dims_data[0] == 0 && i0 == 0) ||
139 (i0 >= 0 && i0 < dims_data[0]));
140 TFLITE_DCHECK((dims_data[1] == 0 && i1 == 0) ||
141 (i1 >= 0 && i1 < dims_data[1]));
142 TFLITE_DCHECK((dims_data[2] == 0 && i2 == 0) ||
143 (i2 >= 0 && i2 < dims_data[2]));
144 TFLITE_DCHECK((dims_data[3] == 0 && i3 == 0) ||
145 (i3 >= 0 && i3 < dims_data[3]));
146 TFLITE_DCHECK((dims_data[4] == 0 && i4 == 0) ||
147 (i4 >= 0 && i4 < dims_data[4]));
148 return (((i0 * dims_data[1] + i1) * dims_data[2] + i2) * dims_data[3] + i3) *
149 dims_data[4] +
150 i4;
151}
