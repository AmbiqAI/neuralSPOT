151//
152// This macro determines a mask given the first and last modules desired. e.g.
153// REGDUMP_MOD_MASK(2,4) // Dump regs for modules 2, 3, and 4
154//
155#define REGDUMP_MOD_MASK(modfirst, modlast) \
156 (((1 << (modlast - modfirst + 1)) - 1) << modfirst)
157
158//
159// These macros determine a single module. e.g.
160// REGDUMP_MOD2 | REGDUMP_MOD4 // Dump regs for modules 2 and 4 (skip 3)
161//
162#define REGDUMP_MOD(n) (1 << n)
163#define REGDUMP_MOD0 (REGDUMP_MOD(0))
164#define REGDUMP_MOD1 (REGDUMP_MOD(1))
