67//*****************************************************************************
68//
69// Option for coreldo and memldo to operate in parallel with simobuck
70// AM_HAL_PWRCTL_SET_CORELDO_MEMLDO_IN_PARALLEL
71// 0 = Do not turn on LDOs in parallel with simobuck.
72// 1 = Turn on LDOs in parallel with simobuck and set their voltage levels
73// ~35mV lower than minimum buck voltages.
74// Default: 1
75//
76//*****************************************************************************
77#define AM_HAL_PWRCTL_SET_CORELDO_MEMLDO_IN_PARALLEL 1
78
79//*****************************************************************************
80//
81// Option to assist VDDC by activating LDOs when disabling SIMOBUCK.
82// AM_HAL_PWRCTRL_LDOS_FOR_VDDC
83// 0 = Do not assist VDDC.
84// 1 = Activate LDOs in parallel when disabling SIMOBUCK.
85// Default: 1
86//
87//*****************************************************************************
88#define AM_HAL_PWRCTRL_LDOS_FOR_VDDC 1
89#if ( (AM_HAL_PWRCTL_SET_CORELDO_MEMLDO_IN_PARALLEL == 0) && (AM_HAL_PWRCTRL_LDOS_FOR_VDDC != 0) )
90#warning AM_HAL_PWRCTRL_LDOS_FOR_VDDC requires AM_HAL_PWRCTL_SET_CORELDO_MEMLDO_IN_PARALLEL.
91#endif
92
93//*****************************************************************************
94//
95// Option for shorting VDDF to VDDS
96// AM_HAL_PWRCTL_SHORT_VDDF_TO_VDDS
97// 0 = Leave inactive.
98// 1 = Set trims to short.
99// Default: 1
100//
101//*****************************************************************************
102#define AM_HAL_PWRCTL_SHORT_VDDF_TO_VDDS 1
103
104//
105// Check for invalid option combinations
106//
107#if ( (AM_HAL_PWRCTL_SET_CORELDO_MEMLDO_IN_PARALLEL != 0) && (AM_HAL_PWRCTL_SHORT_VDDF_TO_VDDS == 0) )
108#error AM_HAL_PWRCTL_SET_CORELDO_MEMLDO_IN_PARALLEL must include AM_HAL_PWRCTL_SHORT_VDDF_TO_VDDS.
