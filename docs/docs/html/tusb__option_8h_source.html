147//--------------------------------------------------------------------
148// RootHub Mode Configuration
149// CFG_TUSB_RHPORTx_MODE contains operation mode and speed for that port
150//--------------------------------------------------------------------
151
152// Lower 4-bit is operational mode
153#define OPT_MODE_NONE 0x00
154#define OPT_MODE_DEVICE 0x01
155#define OPT_MODE_HOST 0x02
156
157// Higher 4-bit is max operational speed (corresponding to tusb_speed_t)
158#define OPT_MODE_FULL_SPEED 0x00
159#define OPT_MODE_LOW_SPEED 0x10
160#define OPT_MODE_HIGH_SPEED 0x20
161
162
163#ifndef CFG_TUSB_RHPORT0_MODE
164 #define CFG_TUSB_RHPORT0_MODE OPT_MODE_NONE
165#endif
166
167
168#ifndef CFG_TUSB_RHPORT1_MODE
169 #define CFG_TUSB_RHPORT1_MODE OPT_MODE_NONE
170#endif
171
172#if (((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_HOST ) && ((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_HOST )) || \
173 (((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_DEVICE) && ((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_DEVICE))
174 #error "TinyUSB currently does not support same modes on more than 1 roothub port"
175#endif
176
177// Which roothub port is configured as host
178#define TUH_OPT_RHPORT ( ((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_HOST) ? 0 : (((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_HOST) ? 1 : -1) )
179#define TUSB_OPT_HOST_ENABLED ( TUH_OPT_RHPORT >= 0 )
180
181// Which roothub port is configured as device
182#define TUD_OPT_RHPORT ( ((CFG_TUSB_RHPORT0_MODE) & OPT_MODE_DEVICE) ? 0 : (((CFG_TUSB_RHPORT1_MODE) & OPT_MODE_DEVICE) ? 1 : -1) )
183
184#if TUD_OPT_RHPORT == 0
185#define TUD_OPT_HIGH_SPEED ( (CFG_TUSB_RHPORT0_MODE) & OPT_MODE_HIGH_SPEED )
186#else
187#define TUD_OPT_HIGH_SPEED ( (CFG_TUSB_RHPORT1_MODE) & OPT_MODE_HIGH_SPEED )
188#endif
189
190#define TUSB_OPT_DEVICE_ENABLED ( TUD_OPT_RHPORT >= 0 )

329//------------- Unaligned Memory -------------//
330
331// ARMv7+ (M3-M7, M23-M33) can access unaligned memory
332#if (defined(__ARM_ARCH) && (__ARM_ARCH >= 7))
333 #define TUP_ARCH_STRICT_ALIGN 0
334#else
335 #define TUP_ARCH_STRICT_ALIGN 1
336#endif
337
338// TUP_MCU_STRICT_ALIGN will overwrite TUP_ARCH_STRICT_ALIGN.
339// In case TUP_MCU_STRICT_ALIGN = 1 and TUP_ARCH_STRICT_ALIGN =0, we will not reply on compiler
340// to generate unaligned access code.
341// LPC_IP3511 Highspeed cannot access unaligned memory on USB_RAM
342#if TUD_OPT_HIGH_SPEED && (CFG_TUSB_MCU == OPT_MCU_LPC54XXX || CFG_TUSB_MCU == OPT_MCU_LPC55XX)
343 #define TUP_MCU_STRICT_ALIGN 1
344#else
345 #define TUP_MCU_STRICT_ALIGN 0
346#endif
