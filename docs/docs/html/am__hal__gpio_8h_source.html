70//
71// Compute some relative offsets needed in the HAL.
72// GPIO_INTX_DELTA: The number of bytes between successive interrupt registers
73// (i.e. INTEN0, INTEN1, INTEN2, ...).
74// GPIO_NXINT_DELTA: The number of bytes between MCUN0INTxx and MCUN1INTxx regs.
75//
76#define GPIO_INTX_DELTA (offsetof(GPIO_Type, MCUN0INT1EN) - offsetof(GPIO_Type, MCUN0INT0EN))
77#define GPIO_NXINT_DELTA (offsetof(GPIO_Type, MCUN1INT0EN) - offsetof(GPIO_Type, MCUN0INT0EN))
78
79//
80// The interrupt register order is: EN, STAT, CLR, SET.
81// GPIO_INTCLR_DELTA: The number of bytes between EN and CLR registers.
82// GPIO_INTSTAT_DELTA: The number of bytes between EN and STAT registers.
83//
84#define GPIO_INTSTAT_DELTA (offsetof(GPIO_Type, MCUN1INT0STAT) - offsetof(GPIO_Type, MCUN0INT0EN))
85#define GPIO_INTCLR_DELTA (offsetof(GPIO_Type, MCUN1INT0CLR) - offsetof(GPIO_Type, MCUN0INT0EN))
86#define GPIO_INTSET_DELTA (offsetof(GPIO_Type, MCUN1INT0SET) - offsetof(GPIO_Type, MCUN0INT0EN))
87
88//
89// Other GPIO helper macros.
90// GPIO_IRQ2N() Given an IRQ, determine N (0 or 1).
91// GPIO_NUM2IDX() Given a GPIO number, determine the interrupt reg index
92// relative to N (0-3).
93// GPIO_NUM2IDX() Given a GPIO number, compute the 32-bit index
94// (e.g. 31=0, 32=1, 63=1, 64=2)
95// GPIO_NUM2MSK() Given a GPIO number, determine the interrupt mask for
96// that bit.
97// GPIO_NUM_IRQS The total number of GPIO IRQs.
98//
99#define GPIO_IRQ2N(irq) ( (irq - GPIO0_001F_IRQn) / (GPIO0_607F_IRQn - GPIO0_001F_IRQn + 1) )
100#define GPIO_IRQ2IDX(irq) ( (irq - GPIO0_001F_IRQn) % (GPIO0_607F_IRQn - GPIO0_001F_IRQn + 1) )
101#define GPIO_NUM2IDX(num) ( num / 32 )
102#define GPIO_NUM2MSK(num) ( 1 << (num & 0x1F) )
103
104#define GPIO_NUM_IRQS (GPIO1_607F_IRQn - GPIO0_001F_IRQn + 1)
