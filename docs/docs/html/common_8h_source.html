16// This file defines common C types and APIs for implementing operations,
17// delegates and other constructs in TensorFlow Lite. The actual operations and
18// delegates can be defined using C++, but the interface between the interpreter
19// and the operations are C.
20//
21// Summary of abstractions
22// TF_LITE_ENSURE - Self-sufficient error checking
23// TfLiteStatus - Status reporting
24// TfLiteIntArray - stores tensor shapes (dims),
25// TfLiteContext - allows an op to access the tensors
26// TfLiteTensor - tensor (a multidimensional array)
27// TfLiteNode - a single node or operation
28// TfLiteRegistration - the implementation of a conceptual operation.
29// TfLiteDelegate - allows delegation of nodes to alternative backends.
30//
31// Some abstractions in this file are created and managed by Interpreter.
32//
33// NOTE: The order of values in these structs are "semi-ABI stable". New values
34// should be added only to the end of structs and never reordered.
35

156// Since we must not depend on any libraries, define a minimal subset of
157// error macros while avoiding names that have pre-conceived meanings like
158// assert and check.
159
160// Try to make all reporting calls through TF_LITE_KERNEL_LOG rather than
161// calling the context->ReportError function directly, so that message strings
162// can be stripped out if the binary size needs to be severely optimized.
163#ifndef TF_LITE_STRIP_ERROR_STRINGS
164#define TF_LITE_KERNEL_LOG(context, ...) \
165 do { \
166 (context)->ReportError((context), __VA_ARGS__); \
167 } while (false)
168
169#define TF_LITE_MAYBE_KERNEL_LOG(context, ...) \
170 do { \
171 if ((context) != nullptr) { \
172 (context)->ReportError((context), __VA_ARGS__); \
173 } \
174 } while (false)
175#else // TF_LITE_STRIP_ERROR_STRINGS
176#define TF_LITE_KERNEL_LOG(context, ...)
177#define TF_LITE_MAYBE_KERNEL_LOG(context, ...)
178#endif // TF_LITE_STRIP_ERROR_STRINGS

180// Check whether value is true, and if not return kTfLiteError from
181// the current function (and report the error string msg).
182#define TF_LITE_ENSURE_MSG(context, value, msg) \
183 do { \
184 if (!(value)) { \
185 TF_LITE_KERNEL_LOG((context), __FILE__ " " msg); \
186 return kTfLiteError; \
187 } \
188 } while (0)
189
190// Check whether the value `a` is true, and if not return kTfLiteError from
191// the current function, while also reporting the location of the error.
192#define TF_LITE_ENSURE(context, a) \
193 do { \
194 if (!(a)) { \
195 TF_LITE_KERNEL_LOG((context), "%s:%d %s was not true.", __FILE__, \
196 __LINE__, #a); \
197 return kTfLiteError; \
198 } \
199 } while (0)
200
201#define TF_LITE_ENSURE_STATUS(a) \
202 do { \
203 const TfLiteStatus s = (a); \
204 if (s != kTfLiteOk) { \
205 return s; \
206 } \
207 } while (0)
208
209// Check whether the value `a == b` is true, and if not return kTfLiteError from
210// the current function, while also reporting the location of the error.
211// `a` and `b` may be evaluated more than once, so no side effects or
212// extremely expensive computations should be done.
213// NOTE: Use TF_LITE_ENSURE_TYPES_EQ if comparing TfLiteTypes.
214#define TF_LITE_ENSURE_EQ(context, a, b) \
215 do { \
216 if ((a) != (b)) { \
217 TF_LITE_KERNEL_LOG((context), "%s:%d %s != %s (%d != %d)", __FILE__, \
218 __LINE__, #a, #b, (a), (b)); \
219 return kTfLiteError; \
220 } \
221 } while (0)
222
223#define TF_LITE_ENSURE_TYPES_EQ(context, a, b) \
224 do { \
225 if ((a) != (b)) { \
226 TF_LITE_KERNEL_LOG((context), "%s:%d %s != %s (%s != %s)", __FILE__, \
227 __LINE__, #a, #b, TfLiteTypeGetName(a), \
228 TfLiteTypeGetName(b)); \
229 return kTfLiteError; \
230 } \
231 } while (0)
232
233#define TF_LITE_ENSURE_NEAR(context, a, b, epsilon) \
234 do { \
235 auto delta = ((a) > (b)) ? ((a) - (b)) : ((b) - (a)); \
236 if (delta > epsilon) { \
237 TF_LITE_KERNEL_LOG((context), "%s:%d %s not near %s (%f != %f)", \
238 __FILE__, __LINE__, #a, #b, static_cast<double>(a), \
239 static_cast<double>(b)); \
240 return kTfLiteError; \
241 } \
242 } while (0)
243
244#define TF_LITE_ENSURE_OK(context, status) \
245 do { \
246 const TfLiteStatus s = (status); \
247 if ((s) != kTfLiteOk) { \
248 return s; \
249 } \
250 } while (0)

326// Memory allocation strategies.
327// * kTfLiteMmapRo: Read-only memory-mapped data, or data externally allocated.
328// * kTfLiteArenaRw: Arena allocated with no guarantees about persistence,
329// and available during eval.
330// * kTfLiteArenaRwPersistent: Arena allocated but persistent across eval, and
331// only available during eval.
332// * kTfLiteDynamic: Allocated during eval, or for string tensors.
333// * kTfLitePersistentRo: Allocated and populated during prepare. This is
334// useful for tensors that can be computed during prepare and treated
335// as constant inputs for downstream ops (also in prepare).
336// * kTfLiteCustom: Custom memory allocation provided by the user. See
337// TfLiteCustomAllocation below.
338typedef enum TfLiteAllocationType {
339 kTfLiteMemNone = 0,
340 kTfLiteMmapRo,
341 kTfLiteArenaRw,
342 kTfLiteArenaRwPersistent,
343 kTfLiteDynamic,
344 kTfLitePersistentRo,
345 kTfLiteCustom,
346} TfLiteAllocationType;

398// A tensor in the interpreter system which is a wrapper around a buffer of
399// data including a dimensionality (or NULL if not currently defined).
400#ifndef TF_LITE_STATIC_MEMORY
401typedef struct TfLiteTensor {
402 // The data type specification for data stored in `data`. This affects
403 // what member of `data` union should be used.
404 TfLiteType type;
405 // A union of data pointers. The appropriate type should be used for a typed
406 // tensor based on `type`.
407 TfLitePtrUnion data;
408 // A pointer to a structure representing the dimensionality interpretation
409 // that the buffer should have. NOTE: the product of elements of `dims`
410 // and the element datatype size should be equal to `bytes` below.
411 TfLiteIntArray* dims;
412 // Quantization information.
413 TfLiteQuantizationParams params;
414 // How memory is mapped
415 // kTfLiteMmapRo: Memory mapped read only.
416 // i.e. weights
417 // kTfLiteArenaRw: Arena allocated read write memory
418 // (i.e. temporaries, outputs).
419 TfLiteAllocationType allocation_type;
420 // The number of bytes required to store the data of this Tensor. I.e.
421 // (bytes of each element) * dims[0] * ... * dims[n-1]. For example, if
422 // type is kTfLiteFloat32 and dims = {3, 2} then
423 // bytes = sizeof(float) * 3 * 2 = 4 * 3 * 2 = 24.
424 size_t bytes;
425
426 // An opaque pointer to a tflite::MMapAllocation
427 const void* allocation;
428
429 // Null-terminated name of this tensor.
430 const char* name;
431
432 // The delegate which knows how to handle `buffer_handle`.
433 // WARNING: This is an experimental interface that is subject to change.
434 struct TfLiteDelegate* delegate;
435
436 // An integer buffer handle that can be handled by `delegate`.
437 // The value is valid only when delegate is not null.
438 // WARNING: This is an experimental interface that is subject to change.
439 TfLiteBufferHandle buffer_handle;
440
441 // If the delegate uses its own buffer (e.g. GPU memory), the delegate is
442 // responsible to set data_is_stale to true.
443 // `delegate->CopyFromBufferHandle` can be called to copy the data from
444 // delegate buffer.
445 // WARNING: This is an // experimental interface that is subject to change.
446 bool data_is_stale;
447
448 // True if the tensor is a variable.
449 bool is_variable;
450
451 // Quantization information. Replaces params field above.
452 TfLiteQuantization quantization;
453
454 // Parameters used to encode a sparse tensor.
455 // This is optional. The field is NULL if a tensor is dense.
456 // WARNING: This is an experimental interface that is subject to change.
457 TfLiteSparsity* sparsity;
458
459 // Optional. Encodes shapes with unknown dimensions with -1. This field is
460 // only populated when unknown dimensions exist in a read-write tensor (i.e.
461 // an input or output tensor). (e.g. `dims` contains [1, 1, 1, 3] and
462 // `dims_signature` contains [1, -1, -1, 3]).
463 const TfLiteIntArray* dims_signature;
464} TfLiteTensor;

842typedef struct TfLiteRegistration {
843 // Initializes the op from serialized data.
844 // Called only *once* for the lifetime of the op, so any one-time allocations
845 // should be made here (unless they depend on tensor sizes).
846 //
847 // If a built-in op:
848 // `buffer` is the op's params data (TfLiteLSTMParams*).
849 // `length` is zero.
850 // If custom op:

852 // `length` is the size of the buffer.

854 // Returns a type-punned (i.e. void*) opaque data (e.g. a primitive pointer

857 // The returned pointer will be stored with the node in the `user_data` field,

860 // function to return `nullptr` and implement the free function to be a no-op.

871 // Returns kTfLiteOk on success.

905// The flags used in `TfLiteDelegate`. Note that this is a bitmask, so the
906// values should be 1, 2, 4, 8, ...etc.
907typedef enum TfLiteDelegateFlags {
908 kTfLiteDelegateFlagsNone = 0,
909 // The flag is set if the delegate can handle dynamic sized tensors.
910 // For example, the output shape of a `Resize` op with non-constant shape
911 // can only be inferred when the op is invoked.
912 // In this case, the Delegate is responsible for calling
913 // `SetTensorToDynamic` to mark the tensor as a dynamic tensor, and calling
914 // `ResizeTensor` when invoking the op.
915 //
916 // If the delegate isn't capable to handle dynamic tensors, this flag need
917 // to be set to false.
918 kTfLiteDelegateFlagsAllowDynamicTensors = 1,
919
920 // This flag can be used by delegates (that allow dynamic tensors) to ensure
921 // applicable tensor shapes are automatically propagated in the case of tensor
922 // resizing.
923 // This means that non-dynamic (allocation_type != kTfLiteDynamic) I/O tensors
924 // of a delegate kernel will have correct shapes before its Prepare() method
925 // is called. The runtime leverages TFLite builtin ops in the original
926 // execution plan to propagate shapes.
927 //
928 // A few points to note:
929 // 1. This requires kTfLiteDelegateFlagsAllowDynamicTensors. If that flag is
930 // false, this one is redundant since the delegate kernels are re-initialized
931 // every time tensors are resized.
932 // 2. Enabling this flag adds some overhead to AllocateTensors(), since extra
933 // work is required to prepare the original execution plan.
934 // 3. This flag requires that the original execution plan only have ops with
935 // valid registrations (and not 'dummy' custom ops like with Flex).
936 // WARNING: This feature is experimental and subject to change.
937 kTfLiteDelegateFlagsRequirePropagatedShapes = 2
938} TfLiteDelegateFlags;
