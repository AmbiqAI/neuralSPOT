40// These are used in the lower 2 bits of a type field to determine the size of
41// the elements (and or size field) of the item pointed to (e.g. vector).
42enum BitWidth {
43 BIT_WIDTH_8 = 0,
44 BIT_WIDTH_16 = 1,
45 BIT_WIDTH_32 = 2,
46 BIT_WIDTH_64 = 3,
47};
48
49// These are used as the upper 6 bits of a type field to indicate the actual
50// type.
51enum Type {
52 FBT_NULL = 0,
53 FBT_INT = 1,
54 FBT_UINT = 2,
55 FBT_FLOAT = 3,
56 // Types above stored inline, types below (except FBT_BOOL) store an offset.
57 FBT_KEY = 4,
58 FBT_STRING = 5,
59 FBT_INDIRECT_INT = 6,
60 FBT_INDIRECT_UINT = 7,
61 FBT_INDIRECT_FLOAT = 8,
62 FBT_MAP = 9,
63 FBT_VECTOR = 10, // Untyped.
64 FBT_VECTOR_INT = 11, // Typed any size (stores no type table).
65 FBT_VECTOR_UINT = 12,
66 FBT_VECTOR_FLOAT = 13,
67 FBT_VECTOR_KEY = 14,
68 // DEPRECATED, use FBT_VECTOR or FBT_VECTOR_KEY instead.
69 // Read test.cpp/FlexBuffersDeprecatedTest() for details on why.
70 FBT_VECTOR_STRING_DEPRECATED = 15,
71 FBT_VECTOR_INT2 = 16, // Typed tuple (no type table, no size field).
72 FBT_VECTOR_UINT2 = 17,
73 FBT_VECTOR_FLOAT2 = 18,
74 FBT_VECTOR_INT3 = 19, // Typed triple (no type table, no size field).
75 FBT_VECTOR_UINT3 = 20,
76 FBT_VECTOR_FLOAT3 = 21,
77 FBT_VECTOR_INT4 = 22, // Typed quad (no type table, no size field).
78 FBT_VECTOR_UINT4 = 23,
79 FBT_VECTOR_FLOAT4 = 24,
80 FBT_BLOB = 25,
81 FBT_BOOL = 26,
82 FBT_VECTOR_BOOL =
83 36, // To Allow the same type of conversion of type to vector type
84
85 FBT_MAX_TYPE = 37
86};
87
88inline bool IsInline(Type t) { return t <= FBT_FLOAT || t == FBT_BOOL; }
89
90inline bool IsTypedVectorElementType(Type t) {
91 return (t >= FBT_INT && t <= FBT_STRING) || t == FBT_BOOL;
92}
93
94inline bool IsTypedVector(Type t) {
95 return (t >= FBT_VECTOR_INT && t <= FBT_VECTOR_STRING_DEPRECATED) ||
96 t == FBT_VECTOR_BOOL;
97}
98
99inline bool IsFixedTypedVector(Type t) {
100 return t >= FBT_VECTOR_INT2 && t <= FBT_VECTOR_FLOAT4;
101}
102
103inline Type ToTypedVector(Type t, size_t fixed_len = 0) {
104 FLATBUFFERS_ASSERT(IsTypedVectorElementType(t));
105 switch (fixed_len) {
106 case 0: return static_cast<Type>(t - FBT_INT + FBT_VECTOR_INT);
107 case 2: return static_cast<Type>(t - FBT_INT + FBT_VECTOR_INT2);
108 case 3: return static_cast<Type>(t - FBT_INT + FBT_VECTOR_INT3);
109 case 4: return static_cast<Type>(t - FBT_INT + FBT_VECTOR_INT4);
110 default: FLATBUFFERS_ASSERT(0); return FBT_NULL;
111 }
112}
113
114inline Type ToTypedVectorElementType(Type t) {
115 FLATBUFFERS_ASSERT(IsTypedVector(t));
116 return static_cast<Type>(t - FBT_VECTOR_INT + FBT_INT);
117}
118
119inline Type ToFixedTypedVectorElementType(Type t, uint8_t *len) {
120 FLATBUFFERS_ASSERT(IsFixedTypedVector(t));
121 auto fixed_type = t - FBT_VECTOR_INT2;
122 *len = static_cast<uint8_t>(fixed_type / 3 +
123 2); // 3 types each, starting from length 2.
124 return static_cast<Type>(fixed_type % 3 + FBT_INT);
125}

132// TODO: can we do this without conditionals using intrinsics or inline asm
133// on some platforms? Given branch prediction the method below should be
134// decently quick, but it is the most frequently executed function.
135// We could do an (unaligned) 64-bit read if we ifdef out the platforms for
136// which that doesn't work (or where we'd read into un-owned memory).
137template<typename R, typename T1, typename T2, typename T4, typename T8>
138R ReadSizedScalar(const uint8_t *data, uint8_t byte_width) {
139 return byte_width < 4
140 ? (byte_width < 2
141 ? static_cast<R>(flatbuffers::ReadScalar<T1>(data))
142 : static_cast<R>(flatbuffers::ReadScalar<T2>(data)))
143 : (byte_width < 8
144 ? static_cast<R>(flatbuffers::ReadScalar<T4>(data))
145 : static_cast<R>(flatbuffers::ReadScalar<T8>(data)));
146}
147
148inline int64_t ReadInt64(const uint8_t *data, uint8_t byte_width) {
149 return ReadSizedScalar<int64_t, int8_t, int16_t, int32_t, int64_t>(
150 data, byte_width);
151}
152
153inline uint64_t ReadUInt64(const uint8_t *data, uint8_t byte_width) {
154 // This is the "hottest" function (all offset lookups use this), so worth
155 // optimizing if possible.
156 // TODO: GCC apparently replaces memcpy by a rep movsb, but only if count is a
157 // constant, which here it isn't. Test if memcpy is still faster than
158 // the conditionals in ReadSizedScalar. Can also use inline asm.
159 // clang-format off
160 #if defined(_MSC_VER) && defined(_M_X64) && !defined(_M_ARM64EC)
161 // This is 64-bit Windows only, __movsb does not work on 32-bit Windows.
162 uint64_t u = 0;
163 __movsb(reinterpret_cast<uint8_t *>(&u),
164 reinterpret_cast<const uint8_t *>(data), byte_width);
165 return flatbuffers::EndianScalar(u);
166 #else
167 return ReadSizedScalar<uint64_t, uint8_t, uint16_t, uint32_t, uint64_t>(
168 data, byte_width);
169 #endif
170 // clang-format on
171}

173inline double ReadDouble(const uint8_t *data, uint8_t byte_width) {
174 return ReadSizedScalar<double, quarter, half, float, double>(data,
175 byte_width);
176}

178inline const uint8_t *Indirect(const uint8_t *offset, uint8_t byte_width) {
179 return offset - ReadUInt64(offset, byte_width);
180}

182template<typename T> const uint8_t *Indirect(const uint8_t *offset) {
183 return offset - flatbuffers::ReadScalar<T>(offset);
184}

186inline BitWidth WidthU(uint64_t u) {
187#define FLATBUFFERS_GET_FIELD_BIT_WIDTH(value, width) \
188 { \
189 if (!((u) & ~((1ULL << (width)) - 1ULL))) return BIT_WIDTH_##width; \
190 }
191 FLATBUFFERS_GET_FIELD_BIT_WIDTH(u, 8);
192 FLATBUFFERS_GET_FIELD_BIT_WIDTH(u, 16);
193 FLATBUFFERS_GET_FIELD_BIT_WIDTH(u, 32);
194#undef FLATBUFFERS_GET_FIELD_BIT_WIDTH
195 return BIT_WIDTH_64;
196}
197
198inline BitWidth WidthI(int64_t i) {
199 auto u = static_cast<uint64_t>(i) << 1;
200 return WidthU(i >= 0 ? u : ~u);
201}
202
203inline BitWidth WidthF(double f) {
204 return static_cast<double>(static_cast<float>(f)) == f ? BIT_WIDTH_32
205 : BIT_WIDTH_64;
206}

220// Object that has a size, obtained either from size prefix, or elsewhere.
221class Sized : public Object {
222 public:
223 // Size prefix.
224 Sized(const uint8_t *data, uint8_t byte_width)
225 : Object(data, byte_width), size_(read_size()) {}
226 // Manual size.
227 Sized(const uint8_t *data, uint8_t byte_width, size_t sz)
228 : Object(data, byte_width), size_(sz) {}
229 size_t size() const { return size_; }
230 // Access size stored in `byte_width_` bytes before data_ pointer.
231 size_t read_size() const {
232 return static_cast<size_t>(ReadUInt64(data_ - byte_width_, byte_width_));
233 }
234
235 protected:
236 size_t size_;
237};

423 // Reads any type as a int64_t. Never fails, does most sensible conversion.
424 // Truncates floats, strings are attempted to be parsed for a number,
425 // vectors/maps return their size. Returns 0 if all else fails.
426 int64_t AsInt64() const {
427 if (type_ == FBT_INT) {
428 // A fast path for the common case.
429 return ReadInt64(data_, parent_width_);
430 } else
431 switch (type_) {
432 case FBT_INDIRECT_INT: return ReadInt64(Indirect(), byte_width_);
433 case FBT_UINT: return ReadUInt64(data_, parent_width_);
434 case FBT_INDIRECT_UINT: return ReadUInt64(Indirect(), byte_width_);
435 case FBT_FLOAT:
436 return static_cast<int64_t>(ReadDouble(data_, parent_width_));
437 case FBT_INDIRECT_FLOAT:
438 return static_cast<int64_t>(ReadDouble(Indirect(), byte_width_));
439 case FBT_NULL: return 0;
440 case FBT_STRING: return flatbuffers::StringToInt(AsString().c_str());
441 case FBT_VECTOR: return static_cast<int64_t>(AsVector().size());
442 case FBT_BOOL: return ReadInt64(data_, parent_width_);
443 default:
444 // Convert other things to int.
445 return 0;
446 }
447 }

452 int16_t AsInt16() const { return static_cast<int16_t>(AsInt64()); }

482 double AsDouble() const {
483 if (type_ == FBT_FLOAT) {
484 // A fast path for the common case.
485 return ReadDouble(data_, parent_width_);
486 } else
487 switch (type_) {
488 case FBT_INDIRECT_FLOAT: return ReadDouble(Indirect(), byte_width_);
489 case FBT_INT:
490 return static_cast<double>(ReadInt64(data_, parent_width_));
491 case FBT_UINT:
492 return static_cast<double>(ReadUInt64(data_, parent_width_));
493 case FBT_INDIRECT_INT:
494 return static_cast<double>(ReadInt64(Indirect(), byte_width_));
495 case FBT_INDIRECT_UINT:
496 return static_cast<double>(ReadUInt64(Indirect(), byte_width_));
497 case FBT_NULL: return 0.0;
498 case FBT_STRING: {
499#if 1
500#if !defined( _MSC_VER)
501#pragma GCC diagnostic push
502#pragma GCC diagnostic ignored "-Wnull-dereference"
503#endif
504 // See b/173239141 for additional context. Patched via
505 // micro/tools/make/flexbuffers_download.sh
506 // Introduce a segfault for an unsupported code path for TFLM.
507 return *(static_cast<double*>(nullptr));
508#if !defined( _MSC_VER)
509#pragma GCC diagnostic pop
510#endif
511#else
512 // This is the original code
513 double d;
514 flatbuffers::StringToNumber(AsString().c_str(), &d);
515 return d;
516#endif
517 }
518 case FBT_VECTOR: return static_cast<double>(AsVector().size());
519 case FBT_BOOL:
520 return static_cast<double>(ReadUInt64(data_, parent_width_));
521 default:
522 // Convert strings and other things to float.
523 return 0;
524 }
525 }

558 // Convert any type to a JSON-like string. strings_quoted determines if
559 // string values at the top level receive "" quotes (inside other values
560 // they always do). keys_quoted determines if keys are quoted, at any level.
561 // TODO(wvo): add further options to have indentation/newlines.
562 void ToString(bool strings_quoted, bool keys_quoted, std::string &s) const {
563 if (type_ == FBT_STRING) {
564 String str(Indirect(), byte_width_);
565 if (strings_quoted) {
566 flatbuffers::EscapeString(str.c_str(), str.length(), &s, true, false);
567 } else {
568 s.append(str.c_str(), str.length());
569 }
570 } else if (IsKey()) {
571 auto str = AsKey();
572 if (keys_quoted) {
573 flatbuffers::EscapeString(str, strlen(str), &s, true, false);
574 } else {
575 s += str;
576 }
577 } else if (IsInt()) {
578 s += flatbuffers::NumToString(AsInt64());
579 } else if (IsUInt()) {
580 s += flatbuffers::NumToString(AsUInt64());
581 } else if (IsFloat()) {
582 s += flatbuffers::NumToString(AsDouble());
583 } else if (IsNull()) {
584 s += "null";
585 } else if (IsBool()) {
586 s += AsBool() ? "true" : "false";
587 } else if (IsMap()) {
588 s += "{ ";
589 auto m = AsMap();
590 auto keys = m.Keys();
591 auto vals = m.Values();
592 for (size_t i = 0; i < keys.size(); i++) {
593 keys[i].ToString(true, keys_quoted, s);
594 s += ": ";
595 vals[i].ToString(true, keys_quoted, s);
596 if (i < keys.size() - 1) s += ", ";
597 }
598 s += " }";
599 } else if (IsVector()) {
600 AppendToString<Vector>(s, AsVector(), keys_quoted);
601 } else if (IsTypedVector()) {
602 AppendToString<TypedVector>(s, AsTypedVector(), keys_quoted);
603 } else if (IsFixedTypedVector()) {
604 AppendToString<FixedTypedVector>(s, AsFixedTypedVector(), keys_quoted);
605 } else if (IsBlob()) {
606 auto blob = AsBlob();
607 flatbuffers::EscapeString(reinterpret_cast<const char *>(blob.data()),
608 blob.size(), &s, true, false);
609 } else {
610 s += "(?)";
611 }
612 }

634 TypedVector AsTypedVector() const {
635 if (IsTypedVector()) {
636 auto tv =
637 TypedVector(Indirect(), byte_width_, ToTypedVectorElementType(type_));

863inline Reference Map::operator[](const char *key) const {
864 auto keys = Keys();
865 // We can't pass keys.byte_width_ to the comparison function, so we have
866 // to pick the right one ahead of time.
867 int (*comp)(const void *, const void *) = nullptr;
868 switch (keys.byte_width_) {
869 case 1: comp = KeyCompare<uint8_t>; break;
870 case 2: comp = KeyCompare<uint16_t>; break;
871 case 4: comp = KeyCompare<uint32_t>; break;
872 case 8: comp = KeyCompare<uint64_t>; break;
873 default: FLATBUFFERS_ASSERT(false); return Reference();
874 }
875 auto res = std::bsearch(key, keys.data_, keys.size(), keys.byte_width_, comp);
876 if (!res) return Reference(nullptr, 1, NullPackedType());
877 auto i = (reinterpret_cast<uint8_t *>(res) - keys.data_) / keys.byte_width_;
878 return (*static_cast<const Vector *>(this))[i];
879}

899// Flags that configure how the Builder behaves.
900// The "Share" flags determine if the Builder automatically tries to pool
901// this type. Pooling can reduce the size of serialized data if there are
902// multiple maps of the same kind, at the expense of slightly slower
903// serialization (the cost of lookups) and more memory use (std::set).
904// By default this is on for keys, but off for strings.
905// Turn keys off if you have e.g. only one map.
906// Turn strings on if you expect many non-unique string values.
907// Additionally, sharing key vectors can save space if you have maps with
908// identical field populations.

910 BUILDER_FLAG_NONE = 0,
911 BUILDER_FLAG_SHARE_KEYS = 1,
912 BUILDER_FLAG_SHARE_STRINGS = 2,
913 BUILDER_FLAG_SHARE_KEYS_AND_STRINGS = 3,
914 BUILDER_FLAG_SHARE_KEY_VECTORS = 4,
915 BUILDER_FLAG_SHARE_ALL = 7,
916};

1129 size_t EndMap(size_t start) {
1130 // We should have interleaved keys and values on the stack.
1131 // Make sure it is an even number:
1132 auto len = stack_.size() - start;
1133 FLATBUFFERS_ASSERT(!(len & 1));
1134 len /= 2;
1135 // Make sure keys are all strings:
1136 for (auto key = start; key < stack_.size(); key += 2) {
1137 FLATBUFFERS_ASSERT(stack_[key].type_ == FBT_KEY);
1138 }
1139 // Now sort values, so later we can do a binary search lookup.
1140 // We want to sort 2 array elements at a time.
1141 struct TwoValue {
1142 Value key;
1143 Value val;
1144 };
1145 // TODO(wvo): strict aliasing?
1146 // TODO(wvo): allow the caller to indicate the data is already sorted
1147 // for maximum efficiency? With an assert to check sortedness to make sure
1148 // we're not breaking binary search.
1149 // Or, we can track if the map is sorted as keys are added which would be
1150 // be quite cheap (cheaper than checking it here), so we can skip this
1151 // step automatically when appliccable, and encourage people to write in
1152 // sorted fashion.
1153 // std::sort is typically already a lot faster on sorted data though.
1154 auto dict = reinterpret_cast<TwoValue *>(stack_.data() + start);
1155 std::sort(
1156 dict, dict + len, [&](const TwoValue &a, const TwoValue &b) -> bool {
1157 auto as = reinterpret_cast<const char *>(buf_.data() + a.key.u_);
1158 auto bs = reinterpret_cast<const char *>(buf_.data() + b.key.u_);
1159 auto comp = strcmp(as, bs);
1160 // We want to disallow duplicate keys, since this results in a
1161 // map where values cannot be found.
1162 // But we can't assert here (since we don't want to fail on
1163 // random JSON input) or have an error mechanism.
1164 // Instead, we set has_duplicate_keys_ in the builder to
1165 // signal this.
1166 // TODO: Have to check for pointer equality, as some sort
1167 // implementation apparently call this function with the same
1168 // element?? Why?
1169 if (!comp && &a != &b) has_duplicate_keys_ = true;
1170 return comp < 0;
1171 });
1172 // First create a vector out of all keys.
1173 // TODO(wvo): if kBuilderFlagShareKeyVectors is true, see if we can share
1174 // the first vector.
1175 auto keys = CreateVector(start, len, 2, true, false);
1176 auto vec = CreateVector(start + 1, len, 2, false, false, &keys);
1177 // Remove temp elements and return map.
1178 stack_.resize(start);
1179 stack_.push_back(vec);
1180 return static_cast<size_t>(vec.u_);
1181 }

1184 // Any map with such keys won't be able to retrieve all values.

1469 BitWidth ElemWidth(size_t buf_size, size_t elem_index) const {
1470 if (IsInline(type_)) {
1471 return min_bit_width_;
1472 } else {
1473 // We have an absolute offset, but want to store a relative offset
1474 // elem_index elements beyond the current buffer end. Since whether
1475 // the relative offset fits in a certain byte_width depends on
1476 // the size of the elements before it (and their alignment), we have
1477 // to test for each size in turn.
1478 for (size_t byte_width = 1;
1479 byte_width <= sizeof(flatbuffers::largest_scalar_t);
1480 byte_width *= 2) {
1481 // Where are we going to write this offset?
1482 auto offset_loc = buf_size +
1483 flatbuffers::PaddingBytes(buf_size, byte_width) +
1484 elem_index * byte_width;
1485 // Compute relative offset.
1486 auto offset = offset_loc - u_;
1487 // Does it fit?
1488 auto bit_width = WidthU(offset);
1489 if (static_cast<size_t>(static_cast<size_t>(1U) << bit_width) ==
1490 byte_width)
1491 return bit_width;
1492 }
1493 FLATBUFFERS_ASSERT(false); // Must match one of the sizes above.
1494 return BIT_WIDTH_64;
1495 }

1550 Value CreateVector(size_t start, size_t vec_len, size_t step, bool typed,
1551 bool fixed, const Value *keys = nullptr) {
1552 FLATBUFFERS_ASSERT(
1553 !fixed ||
1554 typed); // typed=false, fixed=true combination is not supported.
1555 // Figure out smallest bit width we can store this vector with.
1556 auto bit_width = (std::max)(force_min_bit_width_, WidthU(vec_len));
1557 auto prefix_elems = 1;
1558 if (keys) {
1559 // If this vector is part of a map, we will pre-fix an offset to the keys
1560 // to this vector.
1561 bit_width = (std::max)(bit_width, keys->ElemWidth(buf_.size(), 0));
1562 prefix_elems += 2;
1563 }
1564 Type vector_type = FBT_KEY;
1565 // Check bit widths and types for all elements.
1566 for (size_t i = start; i < stack_.size(); i += step) {
1567 auto elem_width =
1568 stack_[i].ElemWidth(buf_.size(), i - start + prefix_elems);
1569 bit_width = (std::max)(bit_width, elem_width);
1570 if (typed) {
1571 if (i == start) {
1572 vector_type = stack_[i].type_;
1573 } else {
1574 // If you get this assert, you are writing a typed vector with
1575 // elements that are not all the same type.
1576 FLATBUFFERS_ASSERT(vector_type == stack_[i].type_);
1577 }
1578 }
1579 }
1580 // If you get this assert, your typed types are not one of:
1581 // Int / UInt / Float / Key.
1582 FLATBUFFERS_ASSERT(!typed || IsTypedVectorElementType(vector_type));
1583 auto byte_width = Align(bit_width);
1584 // Write vector. First the keys width/offset if available, and size.
1585 if (keys) {
1586 WriteOffset(keys->u_, byte_width);
1587 Write<uint64_t>(1ULL << keys->min_bit_width_, byte_width);
1588 }
1589 if (!fixed) Write<uint64_t>(vec_len, byte_width);
1590 // Then the actual data.
1591 auto vloc = buf_.size();
1592 for (size_t i = start; i < stack_.size(); i += step) {
1593 WriteAny(stack_[i], byte_width);
1594 }
1595 // Then the types.
1596 if (!typed) {
1597 for (size_t i = start; i < stack_.size(); i += step) {
1598 buf_.push_back(stack_[i].StoredPackedType(bit_width));
1599 }
1600 }
1601 return Value(static_cast<uint64_t>(vloc),
1602 keys ? FBT_MAP
1603 : (typed ? ToTypedVector(vector_type, fixed ? vec_len : 0)
1604 : FBT_VECTOR),
1605 bit_width);
1606 }

1737 bool VerifyVector(Reference r, const uint8_t *p, Type elem_type) {
1738 // Any kind of nesting goes thru this function, so guard against that
1739 // here, both with simple nesting checks, and the reuse tracker if on.
1740 depth_++;
1741 num_vectors_++;
1742 if (!Check(depth_ <= max_depth_ && num_vectors_ <= max_vectors_))
1743 return false;
1744 auto size_byte_width = r.byte_width_;
1745 FLEX_CHECK_VERIFIED(p, PackedType(Builder::WidthB(size_byte_width), r.type_));
1746 if (!VerifyBeforePointer(p, size_byte_width))
1747 return false;
1748 auto sized = Sized(p, size_byte_width);
1749 auto num_elems = sized.size();
1750 auto elem_byte_width =
1751 r.type_ == FBT_STRING || r.type_ == FBT_BLOB ? uint8_t(1) : r.byte_width_;
1752 auto max_elems = SIZE_MAX / elem_byte_width;
1753 if (!Check(num_elems < max_elems))
1754 return false; // Protect against byte_size overflowing.
1755 auto byte_size = num_elems * elem_byte_width;
1756 if (!VerifyFromPointer(p, byte_size))
1757 return false;
1758 if (elem_type == FBT_NULL) {
1759 // Verify type bytes after the vector.
1760 if (!VerifyFromPointer(p + byte_size, num_elems)) return false;
1761 auto v = Vector(p, size_byte_width);
1762 for (size_t i = 0; i < num_elems; i++)
1763 if (!VerifyRef(v[i])) return false;
1764 } else if (elem_type == FBT_KEY) {
1765 auto v = TypedVector(p, elem_byte_width, FBT_KEY);
1766 for (size_t i = 0; i < num_elems; i++)
1767 if (!VerifyRef(v[i])) return false;
1768 } else {
1769 FLATBUFFERS_ASSERT(IsInline(elem_type));
1770 }
1771 depth_--;
1772 return true;
1773 }

1785 static_cast<uint8_t>(ReadUInt64(p + byte_width, byte_width));
1786 if (!VerifyByteWidth(key_byte_with))

1789 p - off, FBT_KEY);

1808 if (!VerifyByteWidth(r.byte_width_) || !VerifyType(r.type_)) {

1821 return false;

1827 case FBT_KEY:

1829 case FBT_MAP:

1831 VerifyKeys(p, r.byte_width_);
1832 case FBT_VECTOR:

1834 case FBT_VECTOR_INT:
1835 return VerifyVector(r, p, FBT_INT);

1845 return VerifyVector(r, p, FBT_KEY);
1846 case FBT_BLOB:

1848 case FBT_STRING:

1850 VerifyTerminator(String(p, r.byte_width_));
1851 case FBT_VECTOR_INT2:
1852 case FBT_VECTOR_UINT2:
1853 case FBT_VECTOR_FLOAT2:
1854 case FBT_VECTOR_INT3:
1855 case FBT_VECTOR_UINT3:
1856 case FBT_VECTOR_FLOAT3:
1857 case FBT_VECTOR_INT4:
