132static inline void am_hal_usb_hardware_unreset(void)
133{
134 //
135 // Unreset the USB controller and PHY. To make sure the USB can do enumuration, all the USB charger detection
136 // circuit should be disconnected from the Dp/Dm pins.
137 //
138 USB->BCDETCRTL1 =
139 _VAL2FLD(USB_BCDETCRTL1_USBSWRESET, 1) |
140 _VAL2FLD(USB_BCDETCRTL1_USBDCOMPEN, 0) |
141 _VAL2FLD(USB_BCDETCRTL1_USBDCOMPREF, 0) |
142 _VAL2FLD(USB_BCDETCRTL1_IDPSINKEN, 0) |
143 _VAL2FLD(USB_BCDETCRTL1_VDMSRCEN, 0) |
144 _VAL2FLD(USB_BCDETCRTL1_RDMPDWNEN, 0) |
145 _VAL2FLD(USB_BCDETCRTL1_VDPSRCEN, 0) |
146 _VAL2FLD(USB_BCDETCRTL1_IDPSRCEN, 0) |
147 _VAL2FLD(USB_BCDETCRTL1_IDMSINKEN, 0) |
148 _VAL2FLD(USB_BCDETCRTL1_BCWEAKPULLDOWNEN, 0) |
149 _VAL2FLD(USB_BCDETCRTL1_BCWEAKPULLUPEN, 0);

229//*****************************************************************************
230//
242//
243//*****************************************************************************
244static inline am_hal_usb_charger_type_e am_hal_usb_charger_sdp_connection_status(void)
245{
246 //
247 // During Primary Detection the PD shall turn on VDP_SRC and IDM_SINK. When a voltage of VDP_SRC is
248 // applied to Dp, an SDP will continue pulling Dm low through RDM_DWN.
249 // A PD shall compare the voltage on Dm with VDAT_REF. If Dm is less than VDAT_REF, then the PD is
250 // allowed determining that it is attached to an SDP. Because the Dm is less than VDAT_REF, the
251 // CPDETECTED would be zero.
252 //
253 return (USB->BCDETSTATUS_b.CPDETECTED == 0) ? AM_HAL_USB_CHARGER_SDP: AM_HAL_USB_CHARGER_NO_CHARGER;
254}

256//*****************************************************************************
257//
268//
269//*****************************************************************************
270static inline void am_hal_usb_charger_enable_secondary_detection(void)
271{
272 //
273 // Secondary Detection can be used to distinguish between a DCP and a CDP. PDs that are not ready
274 // to be enumerated within TSVLD_CON_PWD of detecting VBUS are required to implement Secondary
275 // Detection. PDs that are ready to be enumerated are allowed to bypass Secondary Detection.
276 // During Secondary Detection, a PD shall output VDM_SRC(0.5V) on Dm, turn on IDP_SINK, and compare the
277 // voltage on Dp to VDAT_REF(0.25V). Since a DCP is required to short Dp to Dm through a resistance
278 // of RDCP_DAT(200 ohm), the voltage on Dp will be close to VDM_SRC, which is above VDAT_REF.
279 // If a PD detects that Dp is greater than VDAT_REF, it knows that it is attached to a DCP.
280 //
281 USB->BCDETCRTL1 =
282 _VAL2FLD(USB_BCDETCRTL1_USBSWRESET, 0) |
283 _VAL2FLD(USB_BCDETCRTL1_VDMSRCEN, 1) |
284 _VAL2FLD(USB_BCDETCRTL1_IDPSINKEN, 1);
285}

288//*****************************************************************************
289//
299//
300//*****************************************************************************
301static inline void am_hal_usb_charger_enable_cdp_charger(void)
302{
303 //
304 // If a PD detects that Dp is less than VDAT_REF, it knows that it is attached to a CDP.
305 // It is then required to turn off VDP_SRC and VDM_SRC, as shown in the Good Battery
306 // Algorithm in Section 3.3.2, and is allowed to draw IDEV_CHG.
307 // Note: Here all bits in BCDETCRTL1 are cleared to zero.
308 //
309 USB->BCDETCRTL1 =
310 _VAL2FLD(USB_BCDETCRTL1_USBSWRESET, 0) |
311 _VAL2FLD(USB_BCDETCRTL1_VDMSRCEN, 0) |
312 _VAL2FLD(USB_BCDETCRTL1_VDPSRCEN, 0);
313}

329static inline void am_hal_usb_charger_enable_dcp_charger(void)
330{
331 //
332 // If a PD detects that Dp is greater than VDAT_REF, it knows that it is attached to a DCP.
333 // It is then required to enable VDP_SRC or pull Dp to VDP_UP through RDP_UP, as defined in
334 // the Good Battery Algorithm in Section 3.3.2.
335 //
336 USB->BCDETCRTL1 =
337 _VAL2FLD(USB_BCDETCRTL1_USBSWRESET, 0) |
338 _VAL2FLD(USB_BCDETCRTL1_VDPSRCEN, 1);
339}
