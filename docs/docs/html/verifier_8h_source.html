26// Helper class to verify the integrity of a FlatBuffer
27class Verifier FLATBUFFERS_FINAL_CLASS {
28 public:
29 Verifier(const uint8_t *buf, size_t buf_len, uoffset_t _max_depth = 64,
30 uoffset_t _max_tables = 1000000, bool _check_alignment = true)
31 : buf_(buf),
32 size_(buf_len),
33 depth_(0),
34 max_depth_(_max_depth),
35 num_tables_(0),
36 max_tables_(_max_tables),
37 upper_bound_(0),
38 check_alignment_(_check_alignment),
39 flex_reuse_tracker_(nullptr) {
40 FLATBUFFERS_ASSERT(size_ < FLATBUFFERS_MAX_BUFFER_SIZE);
41 }

43 // Central location where any verification failures register.
44 bool Check(bool ok) const {
45 // clang-format off
46 #ifdef FLATBUFFERS_DEBUG_VERIFICATION_FAILURE
47 FLATBUFFERS_ASSERT(ok);
48 #endif
49 #ifdef FLATBUFFERS_TRACK_VERIFIER_BUFFER_SIZE
50 if (!ok)
51 upper_bound_ = 0;
52 #endif
53 // clang-format on
54 return ok;
55 }
56
57 // Verify any range within the buffer.
58 bool Verify(size_t elem, size_t elem_len) const {
59 // clang-format off
60 #ifdef FLATBUFFERS_TRACK_VERIFIER_BUFFER_SIZE
61 auto upper_bound = elem + elem_len;
62 if (upper_bound_ < upper_bound)
63 upper_bound_ = upper_bound;
64 #endif
65 // clang-format on
66 return Check(elem_len < size_ && elem <= size_ - elem_len);
67 }
68
69 template<typename T> bool VerifyAlignment(size_t elem) const {
70 return Check((elem & (sizeof(T) - 1)) == 0 || !check_alignment_);
71 }
72
73 // Verify a range indicated by sizeof(T).
74 template<typename T> bool Verify(size_t elem) const {
75 return VerifyAlignment<T>(elem) && Verify(elem, sizeof(T));
76 }
77
78 bool VerifyFromPointer(const uint8_t *p, size_t len) {
79 auto o = static_cast<size_t>(p - buf_);
80 return Verify(o, len);
81 }

83 // Verify relative to a known-good base pointer.
84 bool Verify(const uint8_t *base, voffset_t elem_off, size_t elem_len) const {
85 return Verify(static_cast<size_t>(base - buf_) + elem_off, elem_len);
86 }

89 bool Verify(const uint8_t *base, voffset_t elem_off) const {
90 return Verify(static_cast<size_t>(base - buf_) + elem_off, sizeof(T));

93 // Verify a pointer (may be NULL) of a table type.
94 template<typename T> bool VerifyTable(const T *table) {
95 return !table || table->Verify(*this);
96 }
97
98 // Verify a pointer (may be NULL) of any vector type.
99 template<typename T> bool VerifyVector(const Vector<T> *vec) const {
100 return !vec || VerifyVectorOrString(reinterpret_cast<const uint8_t *>(vec),
101 sizeof(T));
102 }
103
104 // Verify a pointer (may be NULL) of a vector to struct.
105 template<typename T> bool VerifyVector(const Vector<const T *> *vec) const {
106 return VerifyVector(reinterpret_cast<const Vector<T> *>(vec));
107 }
108
109 // Verify a pointer (may be NULL) to string.
110 bool VerifyString(const String *str) const {
111 size_t end;
112 return !str || (VerifyVectorOrString(reinterpret_cast<const uint8_t *>(str),
113 1, &end) &&
114 Verify(end, 1) && // Must have terminator
115 Check(buf_[end] == '\0')); // Terminating byte must be 0.
116 }
117
118 // Common code between vectors and strings.
119 bool VerifyVectorOrString(const uint8_t *vec, size_t elem_size,
120 size_t *end = nullptr) const {
121 auto veco = static_cast<size_t>(vec - buf_);
122 // Check we can read the size field.
123 if (!Verify<uoffset_t>(veco)) return false;
124 // Check the whole array. If this is a string, the byte past the array
125 // must be 0.
126 auto size = ReadScalar<uoffset_t>(vec);
127 auto max_elems = FLATBUFFERS_MAX_BUFFER_SIZE / elem_size;
128 if (!Check(size < max_elems))
129 return false; // Protect against byte_size overflowing.
130 auto byte_size = sizeof(size) + elem_size * size;
131 if (end) *end = veco + byte_size;
132 return Verify(veco, byte_size);
133 }
134
135 // Special case for string contents, after the above has been called.
136 bool VerifyVectorOfStrings(const Vector<Offset<String>> *vec) const {
137 if (vec) {
138 for (uoffset_t i = 0; i < vec->size(); i++) {
139 if (!VerifyString(vec->Get(i))) return false;
140 }
141 }
142 return true;
143 }

145 // Special case for table contents, after the above has been called.
146 template<typename T> bool VerifyVectorOfTables(const Vector<Offset<T>> *vec) {
147 if (vec) {
148 for (uoffset_t i = 0; i < vec->size(); i++) {
149 if (!vec->Get(i)->Verify(*this)) return false;
150 }
151 }
152 return true;
153 }
154
155 __supress_ubsan__("unsigned-integer-overflow") bool VerifyTableStart(
156 const uint8_t *table) {
157 // Check the vtable offset.
158 auto tableo = static_cast<size_t>(table - buf_);
159 if (!Verify<soffset_t>(tableo)) return false;
160 // This offset may be signed, but doing the subtraction unsigned always
161 // gives the result we want.
162 auto vtableo = tableo - static_cast<size_t>(ReadScalar<soffset_t>(table));
163 // Check the vtable size field, then check vtable fits in its entirety.
164 return VerifyComplexity() && Verify<voffset_t>(vtableo) &&
165 VerifyAlignment<voffset_t>(ReadScalar<voffset_t>(buf_ + vtableo)) &&
166 Verify(vtableo, ReadScalar<voffset_t>(buf_ + vtableo));
167 }
168
169 template<typename T>
170 bool VerifyBufferFromStart(const char *identifier, size_t start) {
171 if (identifier && !Check((size_ >= 2 * sizeof(flatbuffers::uoffset_t) &&
172 BufferHasIdentifier(buf_ + start, identifier)))) {
173 return false;
174 }
175
176 // Call T::Verify, which must be in the generated code for this type.
177 auto o = VerifyOffset(start);
178 return o && reinterpret_cast<const T *>(buf_ + start + o)->Verify(*this)
179 // clang-format off
180 #ifdef FLATBUFFERS_TRACK_VERIFIER_BUFFER_SIZE
181 && GetComputedSize()
182 #endif
183 ;
184 // clang-format on
185 }

187 template<typename T>
188 bool VerifyNestedFlatBuffer(const Vector<uint8_t> *buf,
189 const char *identifier) {
190 if (!buf) return true;
191 Verifier nested_verifier(buf->data(), buf->size());
192 return nested_verifier.VerifyBuffer<T>(identifier);
193 }

195 // Verify this whole buffer, starting with root type T.
196 template<typename T> bool VerifyBuffer() { return VerifyBuffer<T>(nullptr); }
197
198 template<typename T> bool VerifyBuffer(const char *identifier) {
199 return VerifyBufferFromStart<T>(identifier, 0);
200 }
201
202 template<typename T> bool VerifySizePrefixedBuffer(const char *identifier) {
203 return Verify<uoffset_t>(0U) &&
204 ReadScalar<uoffset_t>(buf_) == size_ - sizeof(uoffset_t) &&
205 VerifyBufferFromStart<T>(identifier, sizeof(uoffset_t));
206 }
207
208 uoffset_t VerifyOffset(size_t start) const {
209 if (!Verify<uoffset_t>(start)) return 0;
210 auto o = ReadScalar<uoffset_t>(buf_ + start);
211 // May not point to itself.
212 if (!Check(o != 0)) return 0;
213 // Can't wrap around / buffers are max 2GB.
214 if (!Check(static_cast<soffset_t>(o) >= 0)) return 0;
215 // Must be inside the buffer to create a pointer from it (pointer outside
216 // buffer is UB).
217 if (!Verify(start + o, 1)) return 0;
218 return o;
219 }
220
221 uoffset_t VerifyOffset(const uint8_t *base, voffset_t start) const {
222 return VerifyOffset(static_cast<size_t>(base - buf_) + start);
223 }
224
225 // Called at the start of a table to increase counters measuring data
226 // structure depth and amount, and possibly bails out with false if
227 // limits set by the constructor have been hit. Needs to be balanced
228 // with EndTable().
229 bool VerifyComplexity() {
230 depth_++;
231 num_tables_++;
232 return Check(depth_ <= max_depth_ && num_tables_ <= max_tables_);
233 }
234
235 // Called at the end of a table to pop the depth count.
236 bool EndTable() {
237 depth_--;
238 return true;
239 }
240
241 // Returns the message size in bytes
242 size_t GetComputedSize() const {
243 // clang-format off
244 #ifdef FLATBUFFERS_TRACK_VERIFIER_BUFFER_SIZE
245 uintptr_t size = upper_bound_;
246 // Align the size to uoffset_t
247 size = (size - 1 + sizeof(uoffset_t)) & ~(sizeof(uoffset_t) - 1);
248 return (size > size_) ? 0 : size;
249 #else
250 // Must turn on FLATBUFFERS_TRACK_VERIFIER_BUFFER_SIZE for this to work.
251 (void)upper_bound_;
252 FLATBUFFERS_ASSERT(false);
253 return 0;
254 #endif
255 // clang-format on
256 }

258 std::vector<uint8_t> *GetFlexReuseTracker() {
259 return flex_reuse_tracker_;
260 }
261
262 void SetFlexReuseTracker(std::vector<uint8_t> *rt) {
263 flex_reuse_tracker_ = rt;
264 }

266 private:
267 const uint8_t *buf_;
268 size_t size_;
269 uoffset_t depth_;
270 uoffset_t max_depth_;
271 uoffset_t num_tables_;
272 uoffset_t max_tables_;
273 mutable size_t upper_bound_;
274 bool check_alignment_;
275 std::vector<uint8_t> *flex_reuse_tracker_;
276};
